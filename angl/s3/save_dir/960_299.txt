المعلولة لنفس الفلك قد لا تكون الموجودة في الحال فقط  على أنا نقول  هذه الجزئيات المفصلة المعلومة لنفس الفلك هي الموجودة في الحال أو ينضاف إليها ما يتوقع كونها في الاستقبال  فإن قصرتموه على الموجود في الحال بطل إطلاعه على الغيب وإطلاع الأنبياء في اليقظة وسائر الخلق في النوم على ما سيكون في الاستقبال بواسطة ثم بطل مقتضى الدليل فإنه تحكم بأن من عرف الشيء عرف لوازمه وتوابعه حتى لو عرفنا جميع أسباب الأشياء لعرفنا جميع الحوادث المستقبلة وأسباب جميع الحوادث حاضرة في الحال فإنها هي الحركة السماوية ولكن تقتضي المسبب إما بواسطة أو بوسائط كثيرة  ولا في الاستقبال إلى غير نهاية     وإذا تعدى إلى المستقبل لم يكن له آخر فكيف يعرف تفصيل الجزئيات في الاستقبال إلى غير نهاية وكيف
